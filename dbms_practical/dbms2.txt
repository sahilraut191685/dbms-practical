Create a table Employee with attributes emp_id, emp_name, department, and salary. Insert at
least 8–10 records into the table. Write a PL/SQL program using a TRIGGER to ensure that no
employee can be inserted with a salary less than 10,000, and use a CURSOR to display all
employee details along with the total salary of all employees.


-- CREATE TABLE
CREATE TABLE Employee (
    emp_id INT PRIMARY KEY,
    emp_name VARCHAR(50),
    department VARCHAR(50),
    salary INT
);

-- INSERT RECORDS
INSERT INTO Employee VALUES (1, 'Sahil', 'IT', 20000);

INSERT INTO Employee VALUES (4, 'Amit', 'IT', 22000);
INSERT INTO Employee VALUES (5, 'Sneha', 'HR', 17000);
INSERT INTO Employee VALUES (6, 'Rohan', 'Finance', 25000);
INSERT INTO Employee VALUES (7, 'Neha', 'IT', 30000);
INSERT INTO Employee VALUES (8, 'Kiran', 'HR', 12000);

-- TRIGGER TO PREVENT SALARY < 10000
CREATE OR REPLACE TRIGGER trg_salary_check
BEFORE INSERT ON Employee
FOR EACH ROW
BEGIN
    IF :NEW.salary < 10000 THEN
        RAISE_APPLICATION_ERROR(-20001, 'Salary must be at least 10000');
    END IF;
END;
/

-- TEST TRIGGER (THIS WILL FAIL)

-- CURSOR TO DISPLAY EMPLOYEE DETAILS + TOTAL SALARY
DECLARE
    CURSOR emp_cursor IS
        SELECT emp_id, emp_name, department, salary FROM Employee;

    v_id Employee.emp_id%TYPE;
    v_name Employee.emp_name%TYPE;
    v_dept Employee.department%TYPE;
    v_salary Employee.salary%TYPE;

    total_salary INT := 0;
BEGIN
    OPEN emp_cursor;

    LOOP
        FETCH emp_cursor INTO v_id, v_name, v_dept, v_salary;
        EXIT WHEN emp_cursor%NOTFOUND;

        DBMS_OUTPUT.PUT_LINE('ID: ' || v_id ||
                             ' Name: ' || v_name ||
                             ' Dept: ' || v_dept ||
                             ' Salary: ' || v_salary);

        total_salary := total_salary + v_salary;
    END LOOP;

    CLOSE emp_cursor;

    DBMS_OUTPUT.PUT_LINE('Total Salary: ' || total_salary);
END;
/
