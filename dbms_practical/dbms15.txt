Write SQL queries to:
1 Display student names, course names, and marks using INNER JOIN.
2 Display all students including those not enrolled in any course using LEFT JOIN.
3 Display all courses including those with no enrolled students using RIGHT JOIN.
4 Display all possible student–course combinations including unmatched records using
FULL OUTER JOIN (or equivalent).
5 Create a view named Student_Course_View to display Student Name, Course Name,
and Marks.
6 Create another view to display students who have scored more than 75 marks.
7 Retrieve data from both created views.
8 Drop any one of the created views


```sql id="full_setup"
-- CREATE TABLES
CREATE TABLE Students (
    student_id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL
);

CREATE TABLE Courses (
    course_id INT PRIMARY KEY,
    course_name VARCHAR(50) NOT NULL
);

CREATE TABLE Enrollments (
    enroll_id INT PRIMARY KEY,
    student_id INT,
    course_id INT,
    marks INT,
    FOREIGN KEY (student_id) REFERENCES Students(student_id),
    FOREIGN KEY (course_id) REFERENCES Courses(course_id)
);

-- 1. INNER JOIN
SELECT s.name AS student_name, c.course_name, e.marks
FROM Students s
INNER JOIN Enrollments e ON s.student_id = e.student_id
INNER JOIN Courses c ON e.course_id = c.course_id;

-- 2. LEFT JOIN
SELECT s.name AS student_name, c.course_name, e.marks
FROM Students s
LEFT JOIN Enrollments e ON s.student_id = e.student_id
LEFT JOIN Courses c ON e.course_id = c.course_id;

-- 3. RIGHT JOIN
SELECT s.name AS student_name, c.course_name, e.marks
FROM Students s
RIGHT JOIN Enrollments e ON s.student_id = e.student_id
RIGHT JOIN Courses c ON e.course_id = c.course_id;

-- 4. FULL OUTER JOIN
SELECT s.name AS student_name, c.course_name, e.marks
FROM Students s
FULL OUTER JOIN Enrollments e ON s.student_id = e.student_id
FULL OUTER JOIN Courses c ON e.course_id = c.course_id;

-- 5. CREATE VIEW
CREATE VIEW Student_Course_View AS
SELECT s.name AS student_name, c.course_name, e.marks
FROM Students s
JOIN Enrollments e ON s.student_id = e.student_id
JOIN Courses c ON e.course_id = c.course_id;

-- 6. VIEW FOR MARKS > 75
CREATE VIEW High_Scorers AS
SELECT s.name AS student_name, c.course_name, e.marks
FROM Students s
JOIN Enrollments e ON s.student_id = e.student_id
JOIN Courses c ON e.course_id = c.course_id
WHERE e.marks > 75;
```

Database Triggers Implement and test triggers to maintain data integrity in database
Use appropriate data and implement Before and after



-- CREATE TABLE
CREATE TABLE Employee (
    emp_id INT PRIMARY KEY,
    emp_name VARCHAR(50),
    salary INT
);

-- CREATE LOG TABLE (for AFTER trigger)
CREATE TABLE Employee_Log (
    log_id INT AUTO_INCREMENT PRIMARY KEY,
    emp_id INT,
    action VARCHAR(50),
    log_time TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- INSERT SAMPLE DATA
INSERT INTO Employee VALUES (1, 'Sahil', 20000);
INSERT INTO Employee VALUES (2, 'Rahul', 15000);

-- BEFORE INSERT TRIGGER (salary validation)
DELIMITER //
CREATE TRIGGER trg_before_insert
BEFORE INSERT ON Employee
FOR EACH ROW
BEGIN
    IF NEW.salary < 10000 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Salary must be >= 10000';
    END IF;
END;
//
DELIMITER ;

-- AFTER INSERT TRIGGER (log entry)
DELIMITER //
CREATE TRIGGER trg_after_insert
AFTER INSERT ON Employee
FOR EACH ROW
BEGIN
    INSERT INTO Employee_Log(emp_id, action)
    VALUES (NEW.emp_id, 'INSERT');
END;
//
DELIMITER ;

-- BEFORE UPDATE TRIGGER (prevent negative salary)
DELIMITER //
CREATE TRIGGER trg_before_update
BEFORE UPDATE ON Employee
FOR EACH ROW
BEGIN
    IF NEW.salary < 0 THEN
        SIGNAL SQLSTATE '45000'
        SET MESSAGE_TEXT = 'Salary cannot be negative';
    END IF;
END;
//
DELIMITER ;

-- AFTER DELETE TRIGGER (log deletion)
DELIMITER //
CREATE TRIGGER trg_after_delete
AFTER DELETE ON Employee
FOR EACH ROW
BEGIN
    INSERT INTO Employee_Log(emp_id, action)
    VALUES (OLD.emp_id, 'DELETE');
END;
//
DELIMITER ;
