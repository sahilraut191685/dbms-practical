Stored Function with Cursor for Aggregation Logic
Create a reusable function to: (Use suitable data)
1. Iterate through orders
2. Calculate total sales per customer
Store results in another collection customer_total


-- CREATE TABLES
CREATE TABLE Orders (
    order_id INT PRIMARY KEY,
    cust_id INT,
    amount INT
);

CREATE TABLE customer_total (
    cust_id INT,
    total_sales INT
);

-- STORED FUNCTION WITH CURSOR
CREATE OR REPLACE FUNCTION calc_total_sales
RETURN NUMBER
IS
    CURSOR c1 IS
        SELECT cust_id, amount FROM Orders ORDER BY cust_id;

    v_cust_id Orders.cust_id%TYPE;
    v_amount Orders.amount%TYPE;

    prev_id Orders.cust_id%TYPE := NULL;
    total NUMBER := 0;
BEGIN
    OPEN c1;

    LOOP
        FETCH c1 INTO v_cust_id, v_amount;
        EXIT WHEN c1%NOTFOUND;

        IF prev_id IS NULL THEN
            prev_id := v_cust_id;
        END IF;

        IF v_cust_id = prev_id THEN
            total := total + v_amount;
        ELSE
            INSERT INTO customer_total VALUES (prev_id, total);
            prev_id := v_cust_id;
            total := v_amount;
        END IF;
    END LOOP;

    IF prev_id IS NOT NULL THEN
        INSERT INTO customer_total VALUES (prev_id, total);
    END IF;

    CLOSE c1;
    RETURN 1;
END;
/

-- EXECUTE FUNCTION
SELECT calc_total_sales FROM dual;

-- VIEW RESULT
SELECT * FROM customer_total;

Logical conditions (price > 500 AND author='XYZ') , Pattern matching (title LIKE '%SQL%') ,
String functions (LOWER(), CONCAT()) , Set operators (INTERSECT or UNION for reports).


-- CREATE TABLES
CREATE TABLE Members (
    member_id INT PRIMARY KEY,
    name VARCHAR(50),
    email VARCHAR(100)
);

CREATE TABLE Books (
    book_id INT PRIMARY KEY,
    title VARCHAR(100),
    author VARCHAR(50),
    price INT
);

CREATE TABLE Borrow (
    borrow_id INT PRIMARY KEY,
    member_id INT,
    book_id INT,
    bdate DATE,
    FOREIGN KEY (member_id) REFERENCES Members(member_id),
    FOREIGN KEY (book_id) REFERENCES Books(book_id)
);
