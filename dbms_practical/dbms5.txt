-- CREATE TABLES
CREATE TABLE Customer (
    c_id INT PRIMARY KEY,
    name VARCHAR(50) NOT NULL,
    address VARCHAR(100),
    phno VARCHAR(15),
    email VARCHAR(100) UNIQUE
);

CREATE TABLE Product (
    p_id INT PRIMARY KEY,
    pname VARCHAR(50) NOT NULL,
    c_id INT,
    price INT CHECK (price > 0),
    qty INT CHECK (qty >= 1),
    FOREIGN KEY (c_id) REFERENCES Customer(c_id) ON DELETE CASCADE
);

Transactions (txn_id, cust_id, amount, txn_type)

deposits/withdrawals , DELETE inactive customers . Use: Arithmetic operations (balance +
amount) , Logical operators (AND, OR) , Pattern matching (LIKE 'A%') , String functions
(UPPER(), LENGTH()) , Set operators (UNION to combine customer lists).

    cust_id INT PRIMARY KEY,
